## For use in a KVM/QEMU hypervisor on a Red Hat Enterprise Linux machine, one must set a root password and disable the cloud-init service:
virt-customize -a <QCOW2_IMAGE_FILE_NAME> --root-password password:<password> --uninstall cloud-init

## Example virt-install command:
virt-install \
  --name guest1-rhel7 \
  --memory 2048 \
  --vcpus 2 \
  --disk /path/to/imported/disk.qcow2 \
  --import \
  --os-variant rhel7
